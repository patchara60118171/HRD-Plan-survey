-- Run in Supabase SQL Editor
CREATE TABLE IF NOT EXISTS organizations (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  org_code TEXT UNIQUE NOT NULL, org_name_th TEXT NOT NULL, org_name_en TEXT,
  org_type TEXT DEFAULT 'government', contact_email TEXT, is_active BOOLEAN DEFAULT true,
  logo_url TEXT, settings JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS survey_forms (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  form_key TEXT UNIQUE NOT NULL, form_name TEXT NOT NULL,
  form_url_path TEXT NOT NULL, description TEXT,
  is_active BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS org_form_links (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  org_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  form_id UUID REFERENCES survey_forms(id) ON DELETE CASCADE,
  full_url TEXT NOT NULL, is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(), UNIQUE(org_id, form_id)
);
CREATE TABLE IF NOT EXISTS wellbeing_responses (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  org_id UUID REFERENCES organizations(id), org_code TEXT,
  submitted_at TIMESTAMPTZ DEFAULT NOW(), raw_data JSONB DEFAULT '{}'
);

CREATE INDEX IF NOT EXISTS idx_org_code ON organizations(org_code);
CREATE INDEX IF NOT EXISTS idx_ofl_org ON org_form_links(org_id);
CREATE INDEX IF NOT EXISTS idx_wb_sub ON wellbeing_responses(submitted_at);
CREATE INDEX IF NOT EXISTS idx_ch1_sub ON hrd_ch1_responses(submitted_at);
